Convert Fahrenheit to Celsius

This program converts Fahrenheit to Celsius

The main javascript function of the program

function convertFahrenheitToCelsius(_inById){
  var _element = document.getElementById(_inById);
  var _parsed = Number.parseInt(_element.value, 10);
  if(Number.isNaN(_parsed)) {
    setErrorMessage('_msg', 'Incorrect value');
    return;
  }
  var result = Number.parseFloat((_parsed - 32) * (5/9)).toPrecision(6);

}
